        const arr =['小爱同学','在呢','帮我点一份外卖','好的'];
        let index=0;
        setTimeout(()=>{
          console.log('auto',`这句话要渲染到第${index % 2} 个p元素上`,arr[index]);    
          $('.p1').text(arr[index])
          document.addEventListener('click',bindClick);
        },1000);
        
        function bindClick(){
          if(index>=arr.length) return 
          index ++;
          if(index ===arr.length){
            return
          }
          if(index %2===0){
            $('.p1').text(arr[index])
          }else{
            $('.p2').text(arr[index])
          }      
          console.log('click',`这句话要渲染到第${index % 2}个 p元素上`,arr[index]);
        }
